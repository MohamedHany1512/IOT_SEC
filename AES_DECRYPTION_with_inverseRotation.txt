// AES round constant array
const uint8_t Rcon[10] = { 0x01, 0x02, 0x04, 0x08, 0x10, 0x20, 0x40, 0x80, 0x1b, 0x36 };

// SubBytes operation
void invSubBytes(uint8_t state[4][4]) {
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            state[i][j] = inv_sbox[state[i][j]];
        }
    }
}

// ShiftRows operation
void invShiftRows(uint8_t state[4][4]) {
    uint8_t temp;

    // Shift second row
    temp = state[1][3];
    state[1][3] = state[1][2];
    state[1][2] = state[1][1];
    state[1][1] = state[1][0];
    state[1][0] = temp;

    // Shift third row
    temp = state[2][0];
    state[2][0] = state[2][2];
    state[2][2] = temp;
    temp = state[2][1];
    state[2][1] = state[2][3];
    state[2][3] = temp;

    // Shift fourth row
    temp = state[3][0];
    state[3][0] = state[3][1];
    state[3][1] = state[3][2];
    state[3][2] = state[3][3];
    state[3][3] = temp;
}

// Multiplication in GF(2^8) with reducing polynomial x^8 + x^4 + x^3 + x + 1
uint8_t mul(uint8_t a, uint8_t b) {
    uint8_t p = 0;
    uint8_t high_bit_mask = 0x80;
    uint8_t high_bit = 0;
    for (int i = 0; i < 8; ++i) {
        if (b & 1) {
            p ^= a;
        }
        high_bit = a & high_bit_mask;
        a <<= 1;
        if (high_bit) {
            a ^= 0x1b;  // reducing polynomial
        }
        b >>= 1;
    }
    return p;
}

// Inverse MixColumns operation
void invMixColumns(uint8_t state[4][4]) {
    uint8_t tmp[4];
    for (int i = 0; i < 4; ++i) {
        tmp[0] = mul(0x0e, state[0][i]) ^ mul(0x0b, state[1][i]) ^ mul(0x0d, state[2][i]) ^ mul(0x09, state[3][i]);
        tmp[1] = mul(0x09, state[0][i]) ^ mul(0x0e, state[1][i]) ^ mul(0x0b, state[2][i]) ^ mul(0x0d, state[3][i]);
        tmp[2] = mul(0x0d, state[0][i]) ^ mul(0x09, state[1][i]) ^ mul(0x0e, state[2][i]) ^ mul(0x0b, state[3][i]);
        tmp[3] = mul(0x0b, state[0][i]) ^ mul(0x0d, state[1][i]) ^ mul(0x09, state[2][i]) ^ mul(0x0e, state[3][i]);
        for (int j = 0; j < 4; ++j) {
            state[j][i] = tmp[j];
        }
    }
}
// AddRoundKey operation
void addRoundKey(uint8_t state[4][4], const uint8_t* roundKey) {
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            state[j][i] ^= roundKey[i * 4 + j];
        }
    }
}

// Key expansion function

// Rotate word function (circular left shift)
void RotWord(uint8_t* word) {
    uint8_t temp = word[0];
    word[0] = word[1];
    word[1] = word[2];
    word[2] = word[3];
    word[3] = temp;
}

// Substitution function (applies S-box to each byte of a word)
void SubWord(uint8_t* word) {
    for (int i = 0; i < 4; i++) {
        word[i] = sbox[word[i]];
    }
}

// Key expansion function
void KeyExpansion(const uint8_t* key, uint8_t* roundKeys, int Nk) {
    // Copy the key bytes directly into the first Nk words of the expanded key
    for (int i = 0; i < Nk * 4; i++) {
        roundKeys[i] = key[i];
    }

    // Generate the remaining round keys
    for (int i = Nk * 4; i < Nb * (Nr + 1) * 4; i += 4) {
        uint8_t temp[4];
        for (int j = 0; j < 4; j++) {
            temp[j] = roundKeys[i - 4 + j];
        }
        if (i % (Nk * 4) == 0) {
            RotWord(temp);
            SubWord(temp);
            temp[0] ^= Rcon[i / (Nk * 4) - 1];
        }
        else if (Nk > 6 && i % (Nk * 4) == 4) {
            SubWord(temp);
        }
        for (int j = 0; j < 4; j++) {
            roundKeys[i + j] = roundKeys[i - Nk * 4 + j] ^ temp[j];
        }
    }
}

// AES inverse rotation function to rotate the state by 180 degree
void inverseRotateState180(uint8_t state[R][C]) {
    // Reverse the columns of the transposed matrix
    for (int i = 0; i < C; i++) {
        for (int j = 0, k = C - 1; j < k; j++, k--) {
            std::swap(state[j][i], state[k][i]);
        }
    }

    // Transpose the state matrix
    for (int i = 0; i < R; i++) {
        for (int j = i; j < C; j++) {
            std::swap(state[i][j], state[j][i]);
        }
    }
}
// AES decryption function
void invCipher(const uint8_t in[4 * 4], uint8_t out[4 * 4], const uint8_t* roundKeys) {
    uint8_t state[4][4];

    // Initialize state with input
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            state[j][i] = in[i * 4 + j];
        }
    }
    inverseRotateState180(state);
    // Initial round
    addRoundKey(state, roundKeys + Nr * 16);

    // Main rounds
    for (int round = Nr - 1; round >= 1; --round) {
        invShiftRows(state);
        invSubBytes(state);
        addRoundKey(state, roundKeys + round * 16);
        invMixColumns(state);
    }

    // Final round
    invShiftRows(state);
    invSubBytes(state);
    addRoundKey(state, roundKeys);

    // Copy state to output
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            out[i * 4 + j] = state[j][i];
        }
    }
}

int main() {
    uint8_t cipherKey[16] = { 0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0a, 0x0b, 0x0c, 0x0d, 0x0e, 0x0f };
    uint8_t expandedKey[176]; // 11 * 16 bytes for 11 round keys
    // Generate round keys from cipherKey
    KeyExpansion(cipherKey, expandedKey, 4);
    uint8_t ciphertext[16] = { 0x70 ,0xd8 ,0x6a ,0x69 ,0xb4 ,0xcd ,0x7b ,0xc4 ,0xc5 ,0xb7 ,0x04 ,0xe0 ,0x5a ,0x80 ,0x30 ,0xd8 };
    uint8_t decrypted[16];

    // Perform AES decryption
    invCipher(ciphertext, decrypted, expandedKey);

    // Output the decrypted data
    std::cout << "Decrypted Data:" << std::endl;
    for (int i = 0; i < 16; ++i) {
        std::cout << std::hex << std::setw(2) << std::setfill('0') << +decrypted[i] << " ";
        if ((i + 1) % 4 == 0) std::cout << std::endl;
    }

    return 0;
}
